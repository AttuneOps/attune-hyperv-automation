Step 1 - KSH Stop VM

Connect via RDP:

mstsc /admin /v:{hypervhost}

Login as user {hypervhostuser} and open a command prompt.

Execute the following script: 

                
$VM='{targetServer.name}'

Write-Host "Deleting VM : $VM"

$i=0
$found=$false

Get-VM | ForEach-Object {
   if ($_.Name -eq $VM) {
       Write-Host "VM $VM found"
       $found=$true
       return
   }
   $i++
}       

if ($found -eq $true) {
    Write-Host "Checking whether VM $VM is running"
    
    $state = (Get-VM $VM).state
    
    Write-Host "VM state=$state"
    
    if ($state -eq "Running") {
        Write-Host "VM is running, shutting it down".
        Stop-VM -Name $VM
        
        $state = (Get-VM $VM).state
        Write-Host "VM state=$state"
        
        if ($state -eq "Off") {
            Write-Host "VM stopped successfully"
        } else {
            Write-Error "VM stop failed" -ErrorAction Stop
        }
    } else {
        Write-Host "VM is stopped. Skipping."
    }
} else {
    Write-Host "No VM $VM found. Skipping."
}

Step 2 - EVM Create Export Directory

Execute the following script: 

                
$FOLDER = "{hypervExportDirectory}"

if (Test-Path $FOLDER) {
  Write-Host "Export folder $FOLDER exists, good"
} else {
  Write-Host "Export folder $FOLDER deos not exist, creating"
  New-Item $FOLDER -ItemType Directory
  if (Test-Path $FOLDER) {
    Write-Host "Export folder $FOLDER created."
  }
}

Step 3 - EVM Delete Export

Execute the following script: 

                
$VM = '{targetServer.name}'

cd {hypervExportDirectory}

$EXPORT_FILE = "$VM"

if (Test-Path $EXPORT_FILE) {
  Write-Host "Deleting $EXPORT_FILE."
  Remove-Item -Recurse $EXPORT_FILE
} else {
  Write-Host "$EXPORT_FILE doesn't exist, skipping."
}

Step 4 - EVM Export VM

https://learn.microsoft.com/en-us/windows-server/virtualization/hyper-v/deploy/export-and-import-virtual-machines

Execute the following script: 

                
$VM='{targetServer.name}'

Export-VM -Name $VM -Path {hypervExportDirectory}
